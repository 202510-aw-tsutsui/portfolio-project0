// 予約登録のバリデーション処理

FUNCTION createReservation(form)

  IF form.name が空 THEN
    エラー「予約者名は必須です」を表示
    RETURN 入力画面
  END IF

  IF form.date が空 THEN
    エラー「予約日は必須です」を表示
    RETURN 入力画面
  END IF

  IF form.time が空 THEN
    エラー「予約時間は必須です」を表示
    RETURN 入力画面
  END IF

  IF form.peopleCount が 0 以下 THEN
    エラー「人数は1名以上必要です」を表示
    RETURN 入力画面
  END IF

  データを保存
  一覧を再描画

  RETURN 成功

END FUNCTION

FUNCTION filteredReservations(keyword, date, status, sort)

  対象データ = reservations

FUNCTION accessControl()

  // 今回は権限制御未実装

5 ステータス遷移のロジック

状態遷移図 必須
タスクのステータス変化の流れ
        ┌──────────────────────────────────────┐
        │        予約ステータス遷移図           │
        └──────────────────────────────────────┘

                 ┌────────────┐
                 │   仮予約    │
                 └─────┬──────┘
                       │
          ┌────────────┼────────────┐
          ▼            ▼            ▼
     ┌──────────┐ ┌──────────┐ ┌──────────┐
     │予約確定   │ │キャンセル │ │来店済み   │
     └──────────┘ └──────────┘ └──────────┘

※ 仮予約から予約確定・キャンセルへ変更可能
※ 予約確定後は来店済みに変更可能
遷移可否の判定ロジック 必須
状態遷移が許可されるかの判定処理
// 予約ステータス遷移の判定ロジック

FUNCTION canTransition(currentStatus, newStatus)

  allowedTransitions = {
    "仮予約"   → ["予約確定", "キャンセル"],
    "予約確定" → ["来店済み", "キャンセル"],
    "来店済み" → [],
    "キャンセル" → []
  }

  allowed = allowedTransitions[currentStatus]

  IF newStatus が allowed に含まれる THEN
    RETURN true
  ELSE
    RETURN false
  END IF

END FUNCTION
